24 /*****************************************************************************
25  * Macros for manipulating the 'typmod' int. An int32_t used as follows:
26  * Plus/minus = Top bit.
27  * Spare bits = Next 2 bits.
28  * SRID = Next 21 bits.
29  * TYPE = Next 6 bits.
30  * ZM Flags = Bottom 2 bits.
31  *****************************************************************************/
32 
33 /* The following (commented out) definitions are taken from POSTGIS
34 #define TYPMOD_GET_SRID(typmod) ((((typmod) & 0x0FFFFF00) - ((typmod) & 0x10000000)) >> 8)
35 #define TYPMOD_SET_SRID(typmod, srid) ((typmod) = (((typmod) & 0xE00000FF) | ((srid & 0x001FFFFF)<<8)))
36 #define TYPMOD_GET_TYPE(typmod) ((typmod & 0x000000FC)>>2)
37 #define TYPMOD_SET_TYPE(typmod, type) ((typmod) = (typmod & 0xFFFFFF03) | ((type & 0x0000003F)<<2))
38 #define TYPMOD_GET_Z(typmod) ((typmod & 0x00000002)>>1)
39 #define TYPMOD_SET_Z(typmod) ((typmod) = typmod | 0x00000002)
40 #define TYPMOD_GET_M(typmod) (typmod & 0x00000001)
41 #define TYPMOD_SET_M(typmod) ((typmod) = typmod | 0x00000001)
42 #define TYPMOD_GET_NDIMS(typmod) (2+TYPMOD_GET_Z(typmod)+TYPMOD_GET_M(typmod))
43 */
44 
45 /* In order to reuse the above (commented out) macros for manipulating the
46  typmod from POSTGIS we need to shift them to take into account that the
47  first 4 bits are taken for the duration type */
48 
49 #define TYPMOD_DEL_DURATION(typmod) (typmod = typmod >> 4 )
50 #define TYPMOD_SET_DURATION(typmod, durtype) ((typmod) = typmod << 4 | durtype)
